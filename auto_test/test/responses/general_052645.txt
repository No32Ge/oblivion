import random

def guess_number_game():
    number_to_guess = random.randint(0, 100)
    attempts = 0

    print("猜数字游戏开始！我已经想好了一个0到100之间的数字。")

    while True:
        attempts += 1
        user_input = input("请输入你猜的数字：")
        
        try:
            user_guess = int(user_input)
            if user_guess < 0 or user_guess > 100:
                print("请输入一个0到100之间的数字。")
                continue
        except ValueError:
            print("请输入一个有效的整数。")
            continue
        
        if user_guess < number_to_guess:
            print("太低了，再试一次。")
        elif user_guess > number_to_guess:
            print("太高了，再试一次。")
        else:
            print(f"恭喜你！你猜对了数字！你一共尝试了{attempts}次。")
            break

if __name__ == "__main__":
    guess_number_game()
